package Alpha;
sub foo { return 'alpha' };
 
package Bravo;
@ISA = qw(Alpha);
sub foo { return 'bravo' };
 
package Charlie;
@ISA = qw(Bravo);
sub foo { return 'charlie' };
 
Charlie->DUPER::foo; # returns alpha
