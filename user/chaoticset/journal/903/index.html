foreach $newcard (@newcards) {

$cardname = $card =~ /^.+?:/;
if ($newcardname eq $cardname) {
$newstock = $newcard =~ /:.+?:/;
$stock = $card =~ /:.+?:/;
$stock = $stock + $newstock;
$card =~ s/:.+?:/:$stock:/;
}
@cards = (@cards, $newcard);
}
}
}
