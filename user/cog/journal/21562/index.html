#!/usr/bin/perl -w
use HTTP::Proxy;
use HTTP::Proxy::HeaderFilter::simple;

my @proxy = qw( http://localhost:8080/ http://proxy2:8080/ );
my $proxy = HTTP::Proxy->new(@ARGV);

$proxy->push_filter(
    request => HTTP::Proxy::HeaderFilter::simple->new(
        sub {
            shift->proxy->agent->proxy( http => $proxy[ rand @proxy ] );
        }
    )
);

$proxy->start;
