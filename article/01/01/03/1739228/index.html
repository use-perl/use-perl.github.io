open(NAMES, "<", "/var/census/names.txt") or die "Do the census first!";
while ($people{<NAMES>}++) {};
close NAMES;
$sum = 0;
foreach $name (sort keys %people) {
  $n = $people{$name};
  $sum += $n;
  printf "%6d $name", $people{$name};
}
print "=" x 72, "\n";
print "Total: $sum\n";
